백준3009번 네 번째 점
----
https://www.acmicpc.net/problem/3009
----

```c (처음)
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
int main() {
	int p[3][2];
	for (int i = 0; i < 3; i++) {
		scanf("%d%d", &p[i][0], &p[i][1]);
	}
	int ans[2];

	bool v[3];
	memset(v, 0, 3);
	for (int t = 0; t < 2; t++) {
		for (int i = 0; i < 3; i++) {
			for (int j = 0; j < 3; j++) {
				if (i!= j && p[i][t] == p[j][t]) v[i] = 1;
			}
		}
		for (int i = 0; i < 3; i++) {
			if (v[i] != 1) ans[t] = p[i][t];
		}
		memset(v, 0, 3);
	}

	printf("%d %d", ans[0], ans[1]);

	return 0;
}
```

```c (이후)
#include<stdio.h>
#include<string.h>
#include<stdbool.h>
int main() {
	int p[3][2];
	for (int i = 0; i < 3; i++) {
		scanf("%d%d", &p[i][0], &p[i][1]);
	}
	int ans[2];

	for (int t = 0; t < 2; t++) {
		if (p[0][t] == p[1][t]) ans[t] = p[2][t];
		else if (p[0][t] == p[2][t]) ans[t] = p[1][t];
		else ans[t] = p[0][t];
	}

	printf("%d %d", ans[0], ans[1]);

	return 0;
}
```
